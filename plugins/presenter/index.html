Presenter

The presenter plugin will present your Wardrobe instances as a hash.

Example

class Bar
  include Wardrobe
  plugin :presenter

  attribute :string, String
  attribute :time, Time
end

class Foo
  include Wardrobe
  plugin :presenter, time_formatter: ->(time) { time.iso8601 }

  attribute :title, String
  attribute :bar, Bar
end

instance = Foo.new(title: 'A Foo', bar: {string: 'A String', time: '2017-01-01T09:00:00+02:00'})
#=> #<Foo:0x007fc2e9165d60 @title="A Foo", @bar=#<Bar:0x007fc2e915c648 @string="A String", @time=2017-01-01 09:00:00 UTC>>

instance._present
#=> {:title=>"A Foo", :bar=>{:string=>"A String", :time=>"2017-01-01T09:00:00+02:00"}}

attributes = {
  bar: {
    time: nil
  }
}
instance._present(attributes: attributes)
#=> {:bar=>{:time=>"2017-01-01T09:00:00+02:00"}}

instance._present(attributes: attributes, time_formatter: ->(input) { input.utc.iso8601 })
#=> {:bar=>{:time=>"2017-01-01T09:00:00Z"}}
