Validation

Wardrobes validation plugin is inspired by Hanami and Dry.rb.

Example

class User
  include Wardrobe
  plugin :validation
  attribute :id, Integer, validates { gt?(1000) }
  attribute :name, String, validates { min_size?(5) & max_size?(50) }
  attribute :status, Symbol, validates { included_in?(:active, :suspended) }
  attribute :friends, Array[User]
  attribute :interests, Hash[String => Symbol], validates { each_key? { max_size?(2) }}
end

user = User.new(
  id: 521,
  name: 'Foo',
  status: :new,
  friends: [
    {
      id: 45,
      name: 'No',
      status: 'inactive'
    }
  ],
  interests: {
    'architecture' => 'medium',
    :sports => 'low',
    :travel => :high
  }
)._validation_errors
